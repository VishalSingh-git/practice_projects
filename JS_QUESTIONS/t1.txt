let number=36


function createSeries(n){
    
    
    let series=[]
    
    while(n>=0){
        
        series.push(n)
        n=n-2;
        
       if(n>=0){
           series.push(n)
        n=n-4 
       }
        
    }
    return series
}


console.log(createSeries(number))

let string="64280011"

function ascendingNumber(string){
    
    let str=string.split("")

    for(let i=0;i<=str.length;i++){
        for(let j=i+1;j<=str.length-1;j++){
            if(str[i]>str[j]){
                let temp =str[i]
                str[i]=str[j]
                str[j]=temp
            }
        }
       
    }
    
    return str.join("")
    
    
}
console.log(ascendingNumber(string))
